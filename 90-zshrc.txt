function update_k8s_info() {
  if command -v kubectl &>/dev/null; then
    K8S_CLUSTER=$(kubectl config current-context 2>/dev/null)
    K8S_NAMESPACE=$(kubectl config view --minify --output 'jsonpath={..namespace}' 2>/dev/null)
    [[ -z "$K8S_NAMESPACE" ]] && K8S_NAMESPACE="default"
  else
    K8S_CLUSTER=""
    K8S_NAMESPACE=""
  fi
}

precmd_functions+=(update_k8s_info)

# Show local time and New York (EST/EDT) time on the right side of the prompt
function update_time_info() {
  local local_time est_time
  local_time=$(date '+%H:%M:%S %Z')
  est_time=$(TZ="America/New_York" date '+%H:%M:%S %Z')
  RPROMPT="%F{cyan}Local ${local_time}%f | %F{yellow}NY ${est_time}%f"
}

precmd_functions+=(update_time_info)

outsecrets() {
  local secret_yaml lines
  secret_yaml=$(cat)

  lines=$(printf '%s' "$secret_yaml" | yq e '.data | to_entries[] | .key + "\t" + .value' -)

  while IFS=$'\t' read -r key b64value; do
    printf '%s: ' "$key"
    printf '%s' "$b64value" | tr -d '\r\n' | base64 --decode 2>/dev/null | awk '{ sub(/%[[:space:]]*$/, ""); print }'
  done <<< "$lines"
}
